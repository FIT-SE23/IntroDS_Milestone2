\section{Hopf algebras}
\begin{frame}{4. Hopf algebra}
    $(H, \nabla, \eta, \Delta, \epsilon, \sigma)$ is a \textbf{Hopf algebra} over $K$ if:
    \begin{itemize}
        \item $H$ is a \textit{vector space} over $K$
        \item $K$-linear \textit{multiplication} $\nabla: H  \otimes H \to H$ and \textit{unit} $\eta:K \to H$ such that $(H,\nabla, \eta)$ is an algebra over $K$. 
        \item $K$-linear \textit{comultiplication} $\Delta: H \to H \otimes H$ and \textit{counit} $\epsilon: H \to K$ such that $(H, \Delta, \epsilon)$ is a coalgebra over $K$.
        \item $K$-linear \textit{antipode} $\sigma: H \to H$
        \item Compatibility conditions
    \end{itemize}
\end{frame}
